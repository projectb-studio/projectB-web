```typescript
import Link from "next/link";
import Image from "next/image";
import type { Product } from "@/types/database";
import { formatPrice } from "@/lib/utils";
import { cn } from "@/lib/utils";

interface ProductCardProps {
  product: Product;
  variant?: "vertical" | "horizontal";
}

export function ProductCard({ product, variant = "vertical" }: ProductCardProps) {
  const hasSale = product.salePrice && product.salePrice > product.price;

  return (
    <Link href={`/product/${product.slug}`} className="group block">
      {/* 1:1 이미지 컨테이너 */}
      <div className="relative aspect-product overflow-hidden bg-pb-off-white">
        <Image
          src={product.imageUrl}
          alt={product.name}
          fill
          sizes="(max-width: 1024px) 50vw, 25vw"
          className="object-cover transition-transform duration-300 group-hover:scale-105"
        />
        {product.badge && (
          <span
            className={cn(
              "badge absolute top-3 left-3",
              product.badge === "NEW" && "badge-new",
              product.badge === "BEST" && "badge-best",
              product.badge.startsWith("-") && "badge-sale"
            )}
          >
            {product.badge}
          </span>
        )}
      </div>

      {/* 상품 정보 */}
      {variant === "vertical" ? (
        <div className="mt-3">
          <p className="text-xs text-pb-gray tracking-industrial uppercase">
            {product.tag}
          </p>
          <h3 className="text-sm font-medium mt-1 line-clamp-2 leading-snug">
            {product.name}
          </h3>
          <div className="mt-1">
            {hasSale ? (
              <div className="flex items-center gap-2">
                <span className="text-sm text-accent-sale font-medium">
                  {formatPrice(product.price)}
                </span>
                <span className="text-xs text-pb-silver line-through">
                  {formatPrice(product.salePrice!)}
                </span>
              </div>
            ) : (
              <p className="text-sm">{formatPrice(product.price)}</p>
            )}
          </div>
        </div>
      ) : (
        <div className="mt-3 flex items-start justify-between gap-2">
          <div className="min-w-0 flex-1">
            <h3 className="text-sm font-medium line-clamp-2 leading-snug">
              {product.name}
            </h3>
            <p className="text-xs text-pb-gray tracking-industrial uppercase mt-1">
              {product.tag}
            </p>
          </div>
          <div className="flex-shrink-0 text-right">
            {hasSale ? (
              <>
                <p className="text-sm text-accent-sale font-medium">
                  {formatPrice(product.price)}
                </p>
                <p className="text-xs text-pb-silver line-through">
                  {formatPrice(product.salePrice!)}
                </p>
              </>
            ) : (
              <p className="text-sm font-medium">{formatPrice(product.price)}</p>
            )}
          </div>
        </div>
      )}
    </Link>
  );
}
```
